    def __init__(self):

        # 1. Window Styling (Transparent & Always on Top)

        """Calls GPT to get a concise answer based on your resume."""
        try:
            response = OPENAI_CLIENT.chat.completions.create(

                messages=[
                    {"role": "system", "content": f"Briefly answer the interview question based on this resume: {RESUME_TEXT}"},

            return response.choices[0].message.content
        except Exception as e:

        def on_data(transcript: aai.RealtimeTranscript):
            if not transcript.text: return
            if isinstance(transcript, aai.RealtimeFinalTranscript):
                # We have a full question, now ask the AI
                hint = self.get_ai_hint(transcript.text)
                self.root.after(0, self.update_text, f"Q: {transcript.text[:50]}...\nA: {hint}")

        transcriber = aai.RealtimeTranscriber(
            on_data=on_data,
            on_error=lambda e: print(f"Error: {e}"),
            sample_rate=16_000
        )
        transcriber.connect()
        microphone_stream = aai.extras.MicrophoneStream(sample_rate=16_000)
        transcriber.stream(microphone_stream)

if __name__ == "__main__":

import threading
import tkinter as tk
from deepgram import DeepgramClient, LiveTranscriptionEvents, LiveOptions, Microphone
from groq import Groq

# --- API KEYS ---
DEEPGRAM_API_KEY = "YOUR_DEEPGRAM_KEY"
GROQ_API_KEY = "YOUR_GROQ_KEY"
# Paste your resume or key notes here for context
RESUME_CONTEXT = "I am a Python developer with 3 years experience..." 

# Initialize Brain
groq_client = Groq(api_key=GROQ_API_KEY)

class InterviewCopilot:
    def __init__(self):
        # 1. Setup Transparent Overlay Window
        self.root = tk.Tk()
        self.root.title("Copilot")
        self.root.overrideredirect(True)      # Removes top bar/borders
        self.root.attributes("-topmost", True) # Stays on top of Zoom/Teams
        self.root.attributes("-alpha", 0.8)    # Semi-transparent
        self.root.geometry("450x200+100+50")   # Size and position
        self.root.configure(bg="#1a1a1a")

        # 2. Text Display
        self.label = tk.Label(self.root, text="Waiting for audio...", fg="#00FF00", 
                              bg="#1a1a1a", font=("Helvetica", 11), 
                              wraplength=430, justify="left")
        self.label.pack(expand=True, fill="both", padx=10, pady=10)

        # 3. Start Engines
        threading.Thread(target=self.start_audio_stream, daemon=True).start()
        self.root.mainloop()

    def get_ai_hint(self, question):
        """Sends transcript to Groq for a blazing fast answer."""
        try:
            self.root.after(0, self.label.config, {"text": "Thinking...", "fg": "yellow"})
            response = groq_client.chat.completions.create(
                model="llama-3.3-70b-versatile",
                messages=[
                    {"role": "system", "content": f"Briefly answer the interview question in 3 bullet points. Use this context: {RESUME_CONTEXT}"},
                    {"role": "user", "content": question}
                ],
                max_tokens=150
            )
            answer = response.choices[0].message.content
            self.root.after(0, self.label.config, {"text": answer, "fg": "white"})
        except Exception as e:
            self.root.after(0, self.label.config, {"text": f"Brain Error: {e}", "fg": "red"})

    def start_audio_stream(self):
        """Streams audio to Deepgram for real-time transcription."""
        try:
            deepgram = DeepgramClient(DEEPGRAM_API_KEY)
            dg_connection = deepgram.listen.live.v("1")

            def on_transcript(self_dg, result, **kwargs):
                transcript = result.channel.alternatives[0].transcript
                if result.is_final and transcript.strip():
                    # If it's a long sentence/question, ask the brain
                    if len(transcript.split()) > 3:
                        self.get_ai_hint(transcript)

            dg_connection.on(LiveTranscriptionEvents.Transcript, on_transcript)

            options = LiveOptions(model="nova-3", smart_format=True)
            dg_connection.start(options)

            # Start Microphone
            microphone = Microphone(dg_connection.send)
            microphone.start()

        except Exception as e:
            self.root.after(0, self.label.config, {"text": f"Audio Error: {e}", "fg": "red"})

if __name__ == "__main__":
    InterviewCopilot()

import threading
import tkinter as tk
import pyaudiowpatch as pyaudio
from deepgram import DeepgramClient, LiveTranscriptionEvents, LiveOptions
from groq import Groq

# --- CONFIGURATION ---
DEEPGRAM_API_KEY = "YOUR_DEEPGRAM_KEY"
GROQ_API_KEY = "YOUR_GROQ_KEY"
RESUME_CONTEXT = "I am a developer with experience in..."

groq_client = Groq(api_key=GROQ_API_KEY)

class UniversalCopilot:
    def __init__(self):
        self.root = tk.Tk()
        self.root.overrideredirect(True)
        self.root.attributes("-topmost", True)
        self.root.attributes("-alpha", 0.9)
        self.root.geometry("500x250+400+50")
        self.root.configure(bg="#121212")

        self.label = tk.Label(self.root, text="System Ready. Waiting for speech...", 
                              fg="#00FF00", bg="#121212", font=("Arial", 11), 
                              wraplength=480, justify="left")
        self.label.pack(expand=True, fill="both", padx=10)

        threading.Thread(target=self.start_universal_audio, daemon=True).start()
        self.root.mainloop()

    def get_ai_response(self, text):
        """Get instant answer from Groq."""
        try:
            res = groq_client.chat.completions.create(
                model="llama-3.3-70b-versatile",
                messages=[
                    {"role": "system", "content": f"Interview mode. 3 short bullet points. Context: {RESUME_CONTEXT}"},
                    {"role": "user", "content": text}
                ]
            )
            ans = res.choices[0].message.content
            self.root.after(0, self.label.config, {"text": ans, "fg": "white"})
        except Exception as e:
            print(f"Groq Error: {e}")

    def start_universal_audio(self):
        """Captures BOTH Mic and Headphone Loopback."""
        try:
            dg_client = DeepgramClient(DEEPGRAM_API_KEY)
            dg_conn = dg_client.listen.live.v("1")

            def on_message(self_dg, result, **kwargs):
                transcript = result.channel.alternatives[0].transcript
                if result.is_final and len(transcript.split()) > 3:
                    threading.Thread(target=self.get_ai_response, args=(transcript,)).start()

            dg_conn.on(LiveTranscriptionEvents.Transcript, on_message)
            dg_conn.start(LiveOptions(model="nova-3", smart_format=True))

            with pyaudio.PyAudio() as p:
                # 1. Get Default Speaker/Headphone Loopback
                wasapi_info = p.get_host_api_info_by_type(pyaudio.paWASAPI)
                default_speakers = p.get_device_info_by_index(wasapi_info["defaultOutputDevice"])
                
                loopback_idx = None
                for loopback in p.get_loopback_device_info_generator():
                    if default_speakers["name"] in loopback["name"]:
                        loopback_idx = loopback["index"]
                        break

                # 2. Open Loopback Stream (Interviewer)
                def stream_callback(in_data, frame_count, time_info, status):
                    dg_conn.send(in_data)
                    return (None, pyaudio.paContinue)

                stream = p.open(format=pyaudio.paInt16,
                                channels=default_speakers["maxInputChannels"],
                                rate=int(default_speakers["defaultSampleRate"]),
                                input=True,
                                input_device_index=loopback_idx,
                                stream_callback=stream_callback)

                # 3. Open Mic Stream (Your Voice) - Optional if you only want to hear interviewer
                # For simplicity in this 'mini' tool, we prioritize the loopback (interviewer)
                
                while dg_conn.is_active():
                    self.root.update()

        except Exception as e:
            self.root.after(0, self.label.config, {"text": f"Audio Error: {e}", "fg": "red"})

if __name__ == "__main__":
    UniversalCopilot()
